void keep_even(list *L) {
//TODO
    int i, j = 0;
    // 遍历线性表，将偶数移到前面
    for (i = 0; i < L->len; i++) {
        if (L->elem[i] % 2 == 0) {
            L->elem[j++] = L->elem[i];
        }
    }
    // 更新线性表长度
    L->len = j;
}
